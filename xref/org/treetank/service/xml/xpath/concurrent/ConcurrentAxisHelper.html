44  /**
45   * <h1>ConcurrentAxisHelper</h1>
46   * <p>
47   * Is the helper for the ConcurrentAxis and realizes the concurrent evaluation of pipeline steps by decoupling
48   * the given axis from the main thread and storing its results in a blocking queue so establish a
49   * producer-consumer-relationship between the ConcurrentAxis and this one.
50   * </p>
51   * <p>
52   * This axis should only be used and instantiated by the ConcurrentAxis. Find more information on how to use
53   * this framework in the ConcurrentAxis documentation.
54   * </p>
55   */
56  public class ConcurrentAxisHelper implements Runnable {
57  
58      /** Axis that computes the results. */
59      private final AbsAxis mAxis;
60  
61      /**
62       * Queue that stores result keys already computed by this axis. End of the
63       * result sequence is marked by the NULL_NODE_KEY. This is used for
64       * communication with the consumer.
65       */
66      private BlockingQueue<Long> mResults;
67  
68      /**
69       * True, if next() has to be called for the given axis after calling
70       * hasNext().
71       */
72      private final boolean callNext;
73  
74      /**
75       * Bind axis step to transaction. Make sure to create a new ReadTransaction
76       * instead of using the parameter rtx. Because of concurrency every axis has
77       * to have it's own transaction.
78       * 
79       * @param rtx
80       *            Transaction to operate with.
81       */
82      public ConcurrentAxisHelper(final IReadTransaction rtx, final AbsAxis axis,
83          final BlockingQueue<Long> results) {
84          mAxis = axis;
85          mResults = results;
86          callNext =
87              !(mAxis instanceof UnionAxis || mAxis instanceof ExceptAxis
88                  || mAxis instanceof ConcurrentExceptAxis || mAxis instanceof IntersectAxis
89                  || mAxis instanceof LiteralExpr || mAxis instanceof AbsComparator
90                  || mAxis instanceof SequenceAxis || mAxis instanceof XPathAxis
91                  || mAxis instanceof DupFilterAxis || mAxis instanceof ConcurrentUnionAxis || mAxis instanceof ConcurrentIntersectAxis);
92      }
93  
94      public void run() {
95          // Compute all results of the given axis and store the results in the
96          // queue.
97          while (mAxis.hasNext()) {
98              // for some axis next(( has to be called here
99              if (callNext) {
100                 mAxis.next();
101             }
102             try {
103                 // store result in queue as soon as there is space left
104                 // System.out.println("put: " +
105                 // mAxis.getTransaction().getNode().getNodeKey());
106                 mResults.put(mAxis.getTransaction().getNode().getNodeKey());
107                 // wait until next thread arrives and exchange blocking queue
108             } catch (final InterruptedException mExp) {
109                 mExp.printStackTrace();
110 
111             }
112         }
113 
114         try {
115             // Mark end of result sequence by the NULL_NODE_KEY
116             mResults.put((Long)EFixed.NULL_NODE_KEY.getStandardProperty());
117         } catch (final InterruptedException mExp) {
118             mExp.printStackTrace();
119         }
120 
121     }
