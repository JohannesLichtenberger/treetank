111     /**
112      * Hidden constructor.
113      *

115      *            Database for centralized operations on related sessions.

123     protected Session(final Database paramDatabase, final ResourceConfiguration paramResourceConf,
124         final SessionConfiguration paramSessionConf) throws AbsTTException {
125         mDatabase = paramDatabase;
126         mResourceConfig = paramResourceConf;
127         mSessionConfig = paramSessionConf;
128         mTransactionMap = new ConcurrentHashMap<Long, IReadTransaction>();
129         mWriteTransactionStateMap = new ConcurrentHashMap<Long, WriteTransactionState>();
130         mSyncTransactionsReturns = new ConcurrentHashMap<Long, Map<Long, Collection<Future<Void>>>>();
131 
132         mTransactionIDCounter = new AtomicLong();
133         mCommitLock = new ReentrantLock(false);
134 
135         // Init session members.
136         mWriteSemaphore = new Semaphore(paramSessionConf.mWtxAllowed);
137         mReadSemaphore = new Semaphore(paramSessionConf.mRtxAllowed);
138 
139         mFac = EStorage.getStorage(mResourceConfig);
140         if (!mFac.exists()) {
141             // Bootstrap uber page and make sure there already is a root
142             // node.
143             mLastCommittedUberPage = new UberPage();
144         } else {
145             final IReader reader = mFac.getReader();
146             final PageReference firstRef = reader.readFirstReference();
147             mLastCommittedUberPage = (UberPage)firstRef.getPage();
148             reader.close();
149         }
150         mClosed = false;
151     }

169     /**
170      * {@inheritDoc}
171      */
172     @Override
173     public synchronized IReadTransaction beginReadTransaction(final long paramRevisionKey,
174         final IItemList mItemList) throws AbsTTException {
175         assertAccess(paramRevisionKey);
176         // Make sure not to exceed available number of read transactions.
177         try {
178             mReadSemaphore.acquire();
179         } catch (final InterruptedException exc) {
180             throw new TTThreadedException(exc);
181         }
182 
183         IReadTransaction rtx = null;
184         // Create new read transaction.
185         rtx =
186             new ReadTransaction(this, mTransactionIDCounter.incrementAndGet(), new ReadTransactionState(this,
187                 mLastCommittedUberPage, paramRevisionKey, mItemList, mFac.getReader()));
188 
189         // Remember transaction for debugging and safe close.
190         if (mTransactionMap.put(rtx.getTransactionID(), rtx) != null) {
191             throw new TTUsageException("ID generation is bogus because of duplicate ID.");
192         }
193         return rtx;
194     }

204     /**
205      * {@inheritDoc}
206      */
207     @Override
208     public synchronized IWriteTransaction beginWriteTransaction(final int paramMaxNodeCount,
209         final int paramMaxTime) throws AbsTTException {
210         assertAccess(mLastCommittedUberPage.getRevision());
211 
212         // Make sure not to exceed available number of write transactions.
213         if (mWriteSemaphore.availablePermits() == 0) {
214             throw new IllegalStateException("There already is a running exclusive write transaction.");
215         }
216         try {
217             mWriteSemaphore.acquire();
218         } catch (final InterruptedException exc) {
219             throw new TTThreadedException(exc);
220 
221         }
222 
223         final long currentID = mTransactionIDCounter.incrementAndGet();
224         final WriteTransactionState wtxState =
225             createWriteTransactionState(currentID, mLastCommittedUberPage.getRevisionNumber(),
226                 mLastCommittedUberPage.getRevisionNumber());
227 
228         // Create new write transaction.
229         final IWriteTransaction wtx =
230             new WriteTransaction(currentID, this, wtxState, paramMaxNodeCount, paramMaxTime);
231 
232         // Remember transaction for debugging and safe close.
233         if (mTransactionMap.put(currentID, wtx) != null
234             || mWriteTransactionStateMap.put(currentID, wtxState) != null) {
235             throw new TTThreadedException("ID generation is bogus because of duplicate ID.");
236         }
237 
238         return wtx;
239 
240     }

244         final IWriter writer = mFac.getWriter();

250     /**
251      * {@inheritDoc}
252      */
253     @Override
254     public synchronized void close() throws AbsTTException {
255         if (!mClosed) {
256             // Forcibly close all open transactions.
257             for (final IReadTransaction rtx : mTransactionMap.values()) {
258                 if (rtx instanceof IWriteTransaction) {
259                     ((IWriteTransaction)rtx).abort();
260                 }
261                 rtx.close();
262             }
263 
264             // Immediately release all ressources.
265             mLastCommittedUberPage = null;
266             mTransactionMap.clear();
267             mWriteTransactionStateMap.clear();
268 
269             mFac.close();
270             mDatabase.removeSession(mResourceConfig.mPath);
271             mClosed = true;
272         }
273     }

275     /**
276      * Checks for valid revision.
277      * 
278      * @param paramRevision
279      *            revision parameter to check
280      * @throws IllegalArgumentException
281      *             if revision isn't valid
282      */
283     protected void assertAccess(final long paramRevision) {
284         if (mClosed) {
285             throw new IllegalStateException("Session is already closed.");
286         }
287         if (paramRevision < 0) {
288             throw new IllegalArgumentException("Revision must be at least 0");
289         } else if (paramRevision > mLastCommittedUberPage.getRevision()) {
290             throw new IllegalArgumentException(new StringBuilder("Revision must not be bigger than").append(
291                 Long.toString(mLastCommittedUberPage.getRevision())).toString());
292         }
293     }

295     protected void closeWriteTransaction(final long mTransactionID) {
296         // Purge transaction from internal state.
297         mTransactionMap.remove(mTransactionID);
298         // Removing the write from the own internal mapping
299         mWriteTransactionStateMap.remove(mTransactionID);
300         // Make new transactions available.
301         mWriteSemaphore.release();
302     }

305         // Purge transaction from internal state.

332     protected synchronized void syncLogs(final NodePageContainer mContToSync, final long mTransactionId)
333         throws TTThreadedException {
334         final ExecutorService exec = Executors.newCachedThreadPool();
335         final Collection<Future<Void>> returnVals = new ArrayList<Future<Void>>();
336         for (final Long key : mWriteTransactionStateMap.keySet()) {
337             if (key != mTransactionId) {
338                 returnVals.add(exec.submit(new LogSyncer(mWriteTransactionStateMap.get(key), mContToSync)));
339             }
340         }
341         exec.shutdown();
342         if (!mSyncTransactionsReturns.containsKey(mTransactionId)) {
343             mSyncTransactionsReturns.put(mTransactionId,
344                 new ConcurrentHashMap<Long, Collection<Future<Void>>>());
345         }
346 
347         if (mSyncTransactionsReturns.get(mTransactionId).put(mContToSync.getComplete().getNodePageKey(),
348             returnVals) != null) {
349             throw new TTThreadedException(
350                 "only one commit and therefore sync per id and nodepage is allowed!");
351         }
352 
353     }

355     protected synchronized void waitForFinishedSync(final long mTransactionKey) throws TTThreadedException {
356         final Map<Long, Collection<Future<Void>>> completeVals =
357             mSyncTransactionsReturns.remove(mTransactionKey);
358         if (completeVals != null) {
359             for (final Collection<Future<Void>> singleVals : completeVals.values()) {
360                 for (final Future<Void> returnVal : singleVals) {
361                     try {
362                         returnVal.get();
363                     } catch (final InterruptedException exc) {
364                         throw new TTThreadedException(exc);
365                     } catch (final ExecutionException exc) {
366                         throw new TTThreadedException(exc);
367                     }
368                 }
369             }
370         }
371     }
