122     /**
123      * Prepare a node for modification. This is getting the node from the
124      * (persistence) layer, storing the page in the cache and setting up the
125      * node for upcoming modification. Note that this only occurs for
126      * {@link INode}s.
127      * 
128      * @param paramNodeKey
129      *            key of the node to be modified
130      * @return an {@link INode} instance
131      * @throws TTIOException
132      *             if IO Error
133      */
134     protected INode prepareNodeForModification(final long paramNodeKey)
135             throws TTIOException {
136         if (paramNodeKey < 0) {
137             throw new IllegalArgumentException("paramNodeKey must be >= 0!");
138         }
139         if (mNodePageCon != null) {
140             throw new IllegalStateException(
141                     "Another node page container is currently in the cache for updates!");
142         }
143 
144         final long nodePageKey = nodePageKey(paramNodeKey);
145         final int nodePageOffset = nodePageOffset(paramNodeKey);
146         prepareNodePage(nodePageKey);
147 
148         INode node = mNodePageCon.getModified().getNode(nodePageOffset);
149         if (node == null) {
150             final INode oldNode = mNodePageCon.getComplete().getNode(
151                     nodePageOffset);
152             if (oldNode == null) {
153                 throw new TTIOException("Cannot retrieve node from cache");
154             }
155             node = oldNode;
156             mNodePageCon.getModified().setNode(nodePageOffset, node);
157         }
158         return node;
159     }

161     /**
162      * Finishing the node modification. That is storing the node including the
163      * page in the cache.
164      * 
165      * @param paramNode
166      *            the node to be modified
167      */
168     protected void finishNodeModification(final INode paramNode) {
169         final long nodePageKey = nodePageKey(paramNode.getNodeKey());
170         if (mNodePageCon == null || paramNode == null
171                 || mLog.get(nodePageKey) == null) {
172             throw new IllegalStateException();
173         }
174 
175         mLog.put(nodePageKey, mNodePageCon);
176 
177         this.mNodePageCon = null;
178 
179     }

192         // Allocate node key and increment node count.

194         final long nodeKey = mNewRoot.getMaxNodeKey();
195         final long nodePageKey = nodePageKey(nodeKey);
196         final int nodePageOffset = nodePageOffset(nodeKey);
197         prepareNodePage(nodePageKey);
198         final NodePage page = mNodePageCon.getModified();
199         page.setNode(nodePageOffset, paramNode);
200         finishNodeModification(paramNode);
201 
202         return paramNode;

262     /**
263      * Removing a node from the storage.
264      * 
265      * @param paramNode
266      *            {@link INode} to be removed
267      * @throws TTIOException
268      *             if the removal fails
269      */
270     protected void removeNode(final INode paramNode) throws TTIOException {
271         assert paramNode != null;
272         final long nodePageKey = nodePageKey(paramNode.getNodeKey());
273         prepareNodePage(nodePageKey);
274         final INode delNode = new DeletedNode(new NodeDelegate(
275                 paramNode.getNodeKey(), paramNode.getParentKey(),
276                 paramNode.getHash()));
277         mNodePageCon.getModified().setNode(
278                 nodePageOffset(paramNode.getNodeKey()), delNode);
279         mNodePageCon.getComplete().setNode(
280                 nodePageOffset(paramNode.getNodeKey()), delNode);
281         finishNodeModification(paramNode);
282     }

287     @Override
288     protected INode getNode(final long mNodeKey) throws TTIOException {
289 
290         // Calculate page and node part for given nodeKey.
291         final long nodePageKey = nodePageKey(mNodeKey);
292         final int nodePageOffset = nodePageOffset(mNodeKey);
293 
294         final NodePageContainer pageCont = mLog.get(nodePageKey);
295         if (pageCont == null) {
296             return super.getNode(mNodeKey);
297         } else if (pageCont.getModified().getNode(nodePageOffset) == null) {
298             final INode item = pageCont.getComplete().getNode(nodePageOffset);
299             return checkItemIfDeleted(item);
300 
301         } else {
302             final INode item = pageCont.getModified().getNode(nodePageOffset);
303             return checkItemIfDeleted(item);
304         }

324             returnVal = super.getName(mNameKey);

354     /**
355      * Committing a a writetransaction. This method is recursivly invoked by all
356      * {@link PageReference}s.
357      * 
358      * @param reference
359      *            to be commited
360      * @throws AbsTTException
361      *             if the write fails
362      */
363     public void commit(final PageReference reference) throws AbsTTException {
364         IPage page = null;
365 
366         // if reference is not null, get one from the persistent storage.
367         if (reference != null) {
368             // first, try to get one from the log
369             final NodePageContainer cont = mLog.get(reference.getNodePageKey());
370             if (cont != null) {
371                 page = cont.getModified();
372             }
373             // if none is in the log, test if one is instantiated, if so, get
374             // the one flexible from the
375             // reference
376             if (page == null) {
377                 page = reference.getPage();
378                 if (page == null) {
379                     return;
380                 }
381             }
382 
383             reference.setPage(page);
384             // Recursively commit indirectely referenced pages and then
385             // write self.
386             page.commit(this);
387             mPageWriter.write(reference);
388             reference.setPage(null);
389             // afterwards synchronize all logs since the changes must to be
390             // written to the transaction log as
391             // well
392             if (cont != null) {
393                 mSession.syncLogs(cont, mTransactionID);
394             }
395         }
396     }

490     protected IndirectPage prepareIndirectPage(
491             final PageReference paramReference) throws TTIOException {
492 
493         IndirectPage page = (IndirectPage) paramReference.getPage();

508     protected NodePageContainer prepareNodePage(final long paramNodePageKey)
509             throws TTIOException {
510 
511         // Last level points to node nodePageReference.
512         NodePageContainer cont = mLog.get(paramNodePageKey);
513         if (cont == null) {
514 
515             // Indirect reference.
516             final PageReference reference = prepareLeafOfTree(
517                     mNewRoot.getIndirectPageReference(), paramNodePageKey);
518             NodePage page = (NodePage) reference.getPage();
519 
520             if (page == null) {
521                 if (reference.getKey() == null) {
522                     cont = new NodePageContainer(new NodePage(paramNodePageKey,
523                             IConstants.UBP_ROOT_REVISION_NUMBER));
524                 } else {
525                     cont = dereferenceNodePageForModification(paramNodePageKey);
526                 }
527             } else {
528                 cont = new NodePageContainer(page);
529             }
530 
531             reference.setNodePageKey(paramNodePageKey);
532             mLog.put(paramNodePageKey, cont);
533         }
534         mNodePageCon = cont;
535         return cont;
536     }

538     private RevisionRootPage preparePreviousRevisionRootPage(
539             final long mBaseRevision, final long representRevision)
540             throws TTIOException {
541 
542         if (getUberPage().isBootstrap()) {
543             return super.loadRevRoot(mBaseRevision);
544         } else {
545 
546             // Prepare revision root nodePageReference.
547             final RevisionRootPage revisionRootPage = new RevisionRootPage(
548                     super.loadRevRoot(mBaseRevision), representRevision + 1);
549 
550             // Prepare indirect tree to hold reference to prepared revision root
551             // nodePageReference.
552             final PageReference revisionRootPageReference = prepareLeafOfTree(
553                     getUberPage().getIndirectPageReference(), getUberPage()
554                             .getRevisionNumber());
555 
556             // Link the prepared revision root nodePageReference with the
557             // prepared indirect tree.
558             revisionRootPageReference.setPage(revisionRootPage);
559 
560             revisionRootPage.getNamePageReference().setPage(
561                     (NamePage) super.getActualRevisionRootPage()
562                             .getNamePageReference().getPage());
563 
564             // Return prepared revision root nodePageReference.
565             return revisionRootPage;
566         }
567     }

569     protected PageReference prepareLeafOfTree(
570             final PageReference mStartReference, final long mKey)
571             throws TTIOException {
572 
573         // Initial state pointing to the indirect nodePageReference of level 0.
574 
575         PageReference reference = mStartReference;
576         int offset = 0;
577         long levelKey = mKey;
578 
579         // Iterate through all levels.
580         for (int level = 0, height = IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT.length; level < height; level++) {
581             offset = (int) (levelKey >> IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[level]);
582             levelKey -= offset << IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[level];
583             final IndirectPage page = prepareIndirectPage(reference);
584             reference = page.getReferences()[offset];
585 
586         }
587 
588         // Return reference to leaf of indirect tree.
589         return reference;
590     }
