250     /**
251      * Adapting everything for move operations.
252      * 
253      * @param paramFromNode
254      *            root {@link IStructNode} of the subtree to be moved
255      * @param paramToNode
256      *            the {@link IStructNode} which is the anchor of the new
257      *            subtree
258      * @param paramInsert
259      *            determines if it has to be inserted as a first child or a
260      *            right sibling
261      * @throws AbsTTException
262      *             if removing a node fails after merging text nodes
263      */
264     private void adaptForMove(final IStructNode paramFromNode, final IStructNode paramToNode,
265         final EInsert paramInsert) throws AbsTTException {
266         assert paramFromNode != null;
267         assert paramToNode != null;
268         assert paramInsert != null;
269 
270         // Modify nodes where the subtree has been moved from.
271         // ==============================================================================
272         final IStructNode parent =
273             (IStructNode)getTransactionState().prepareNodeForModification(paramFromNode.getParentKey());
274         parent.decrementChildCount();
275         // Adapt first child key of former parent.
276         if (parent.getFirstChildKey() == paramFromNode.getNodeKey()) {
277             parent.setFirstChildKey(paramFromNode.getRightSiblingKey());
278         }
279         getTransactionState().finishNodeModification(parent);
280 
281         // Adapt right sibling key of former left sibling.
282         if (paramFromNode.hasLeftSibling()) {
283             final IStructNode leftSibling =
284                 (IStructNode)getTransactionState().prepareNodeForModification(
285                     paramFromNode.getLeftSiblingKey());
286             leftSibling.setRightSiblingKey(paramFromNode.getRightSiblingKey());
287             getTransactionState().finishNodeModification(leftSibling);
288         }
289 
290         // Adapt left sibling key of former right sibling.
291         if (paramFromNode.hasRightSibling()) {
292             final IStructNode rightSibling =
293                 (IStructNode)getTransactionState().prepareNodeForModification(
294                     paramFromNode.getRightSiblingKey());
295             rightSibling.setLeftSiblingKey(paramFromNode.getLeftSiblingKey());
296             getTransactionState().finishNodeModification(rightSibling);
297         }
298 
299         // Merge text nodes.
300         if (paramFromNode.hasLeftSibling() && paramFromNode.hasRightSibling()) {
301             moveTo(paramFromNode.getLeftSiblingKey());
302             if (getCurrentNode() != null && getCurrentNode().getKind() == ENode.TEXT_KIND) {
303                 final StringBuilder builder = new StringBuilder(getValueOfCurrentNode());
304                 moveTo(paramFromNode.getRightSiblingKey());
305                 if (getCurrentNode() != null && getCurrentNode().getKind() == ENode.TEXT_KIND) {
306                     builder.append(getValueOfCurrentNode());
307                     remove();
308                     moveTo(paramFromNode.getLeftSiblingKey());
309                     setValue(builder.toString());
310                 }
311             }
312         }
313 
314         // Modify nodes where the subtree has been moved to.
315         // ==============================================================================
316         paramInsert.processMove(paramFromNode, paramToNode, this);
317     }

472     /**
473      * {@inheritDoc}
474      */
475     @Override
476     public synchronized void remove() throws AbsTTException {
477         checkAccessAndCommit();
478         if (getCurrentNode().getKind() == ENode.ROOT_KIND) {
479             throw new TTUsageException("Document root can not be removed.");
480         } else if (getCurrentNode() instanceof IStructNode) {
481             final IStructNode node = (IStructNode)getCurrentNode();
482             // Remove subtree.
483             for (final AbsAxis desc = new DescendantAxis(this, false); desc.hasNext(); desc.next()) {
484                 getTransactionState().removeNode(getCurrentNode());
485             }
486             moveTo(node.getNodeKey());
487             adaptForRemove(node);
488             adaptHashesWithRemove();
489 
490             // Set current node.
491             if (node.hasRightSibling()) {
492                 moveTo(node.getRightSiblingKey());
493             } else if (node.hasLeftSibling()) {
494                 moveTo(node.getLeftSiblingKey());
495             } else {
496                 moveTo(node.getParentKey());
497             }
498         } else if (getCurrentNode().getKind() == ENode.ATTRIBUTE_KIND) {
499             final INode node = getCurrentNode();
500 
501             final ElementNode parent =
502                 (ElementNode)getTransactionState().prepareNodeForModification(node.getParentKey());
503             parent.removeAttribute(node.getNodeKey());
504             getTransactionState().finishNodeModification(parent);
505             adaptHashesWithRemove();
506             moveToParent();
507         } else if (getCurrentNode().getKind() == ENode.NAMESPACE_KIND) {
508             final INode node = getCurrentNode();
509 
510             final ElementNode parent =
511                 (ElementNode)getTransactionState().prepareNodeForModification(node.getParentKey());
512             parent.removeNamespace(node.getNodeKey());
513             getTransactionState().finishNodeModification(parent);
514             adaptHashesWithRemove();
515             moveToParent();
516         }
517     }

597     @Override
598     public void revertTo(final long paramRevision) throws TTUsageException, TTIOException {
599         if (paramRevision < 0) {
600             throw new IllegalArgumentException("paramRevision parameter must be >= 0");
601         }
602         assertNotClosed();
603         mSession.assertAccess(paramRevision);
604         getTransactionState().close();
605         // Reset internal transaction state to new uber page.
606         setTransactionState(mSession.createWriteTransactionState(getTransactionID(), paramRevision,
607             getRevisionNumber() - 1));
608         // Reset modification counter.
609         mModificationCount = 0L;
610         moveToDocumentRoot();
611

614     /**
615      * {@inheritDoc}
616      */
617     @Override
618     public synchronized void commit() throws AbsTTException {
619 
620         assertNotClosed();
621 
622         // Commit uber page.
623         final UberPage uberPage = getTransactionState().commit();
624 
625         // Remember succesfully committed uber page in session state.
626         mSession.setLastCommittedUberPage(uberPage);
627 
628         // Reset modification counter.
629         mModificationCount = 0L;
630 
631         getTransactionState().close();
632         // Reset internal transaction state to new uber page.
633         setTransactionState(mSession.createWriteTransactionState(getTransactionID(), getRevisionNumber(),
634             getRevisionNumber()));
635 
636     }

640      */
641     @Override
642     public synchronized void abort() throws TTIOException {
643 
644         assertNotClosed();
645 
646         // Reset modification counter.
647         mModificationCount = 0L;
648 
649         getTransactionState().close();
650 
651         long revisionToSet = 0;
652         if (!getTransactionState().getUberPage().isBootstrap()) {
653             revisionToSet = getRevisionNumber() - 1;
654         }
655 
656         // Reset internal transaction state to last committed uber page.
657         setTransactionState(mSession.createWriteTransactionState(getTransactionID(), revisionToSet,
658             revisionToSet));
659     }

661     /**
662      * {@inheritDoc}
663      */
664     @Override
665     public synchronized void close() throws AbsTTException {
666         if (!isClosed()) {
667             // Make sure to commit all dirty data.
668             if (mModificationCount > 0) {
669                 throw new TTUsageException("Must commit/abort transaction first");
670             }
671             // Release all state immediately.
672             getTransactionState().close();
673             mSession.closeWriteTransaction(getTransactionID());
674             setTransactionState(null);
675             setCurrentNode(null);
676             // Remember that we are closed.
677             setClosed();
678         }
679     }

697     /**
698      * Adapting everything for insert operations.
699      * 
700      * @param paramNewNode
701      *            pointer of the new node to be inserted
702      * @param paramInsert
703      *            determines the position where to insert
704      * @throws TTIOException
705      *             if anything weird happens
706      */
707     private void adaptForInsert(final INode paramNewNode, final EInsert paramInsert) throws TTIOException {
708         assert paramNewNode != null;
709         assert paramInsert != null;
710 
711         if (paramNewNode instanceof IStructNode) {
712             final IStructNode strucNode = (IStructNode)paramNewNode;
713             final IStructNode parent =
714                 (IStructNode)getTransactionState().prepareNodeForModification(paramNewNode.getParentKey());
715             parent.incrementChildCount();

756         assert paramOldNode != null;
757 
758         // Adapt left sibling node if there is one.
759         if (paramOldNode.hasLeftSibling()) {
760             final IStructNode leftSibling =
761                 (IStructNode)getTransactionState().prepareNodeForModification(
762                     paramOldNode.getLeftSiblingKey());
763             leftSibling.setRightSiblingKey(paramOldNode.getRightSiblingKey());
764             getTransactionState().finishNodeModification(leftSibling);
765         }
766 
767         // Adapt right sibling node if there is one.
768         if (paramOldNode.hasRightSibling()) {
769             final IStructNode rightSibling =
770                 (IStructNode)getTransactionState().prepareNodeForModification(
771                     paramOldNode.getRightSiblingKey());
772             rightSibling.setLeftSiblingKey(paramOldNode.getLeftSiblingKey());
773             getTransactionState().finishNodeModification(rightSibling);
774         }
775 
776         // Adapt parent, if node has now left sibling it is a first child.
777         final IStructNode parent =
778             (IStructNode)getTransactionState().prepareNodeForModification(paramOldNode.getParentKey());
779         if (!paramOldNode.hasLeftSibling()) {
780             parent.setFirstChildKey(paramOldNode.getRightSiblingKey());
781         }
782         parent.decrementChildCount();
783         getTransactionState().finishNodeModification(parent);
784 
785         if (paramOldNode.getKind() == ENode.ELEMENT_KIND) {
786             // removing attributes
787             for (int i = 0; i < ((ElementNode)paramOldNode).getAttributeCount(); i++) {
788                 moveTo(((ElementNode)paramOldNode).getAttributeKey(i));
789                 getTransactionState().removeNode(this.getCurrentNode());
790             }
791             // removing namespaces
792             moveTo(paramOldNode.getNodeKey());
793             for (int i = 0; i < ((ElementNode)paramOldNode).getNamespaceCount(); i++) {
794                 moveTo(((ElementNode)paramOldNode).getNamespaceKey(i));
795                 getTransactionState().removeNode(this.getCurrentNode());
796             }
797         }
798 
799         // Remove old node.
800         getTransactionState().removeNode(paramOldNode);

899      * Adapting the structure with a rolling hash for all ancestors only with
900      * insert.
901      * 
902      * @throws TTIOException
903      *             if anything weird happened
904      */
905     private void postorderAdd() throws TTIOException {
906         // start with hash to add
907         final INode startNode = getCurrentNode();
908         // long for adapting the hash of the parent
909         long hashCodeForParent = 0;
910         // adapting the parent if the current node is no structural one.
911         if (!(getCurrentNode() instanceof IStructNode)) {
912             getTransactionState().prepareNodeForModification(getCurrentNode().getNodeKey());
913             getCurrentNode().setHash(getCurrentNode().hashCode());
914             getTransactionState().finishNodeModification(getCurrentNode());
915             moveTo(getCurrentNode().getParentKey());
916         }
917         // Cursor to root
918         IStructNode cursorToRoot;
919         do {
920             synchronized (getCurrentNode()) {
921                 cursorToRoot =
922                     (IStructNode)getTransactionState().prepareNodeForModification(
923                         getCurrentNode().getNodeKey());
924                 hashCodeForParent = getCurrentNode().hashCode() + hashCodeForParent * PRIME;
925                 // Caring about attributes and namespaces if node is an element.
926                 if (cursorToRoot.getKind() == ENode.ELEMENT_KIND) {
927                     final ElementNode currentElement = (ElementNode)cursorToRoot;
928                     // setting the attributes and namespaces
929                     for (int i = 0; i < ((ElementNode)cursorToRoot).getAttributeCount(); i++) {
930                         moveTo(currentElement.getAttributeKey(i));
931                         hashCodeForParent = getCurrentNode().hashCode() + hashCodeForParent * PRIME;
932                     }
933                     for (int i = 0; i < ((ElementNode)cursorToRoot).getNamespaceCount(); i++) {
934                         moveTo(currentElement.getNamespaceKey(i));
935                         hashCodeForParent = getCurrentNode().hashCode() + hashCodeForParent * PRIME;
936                     }
937                     moveTo(cursorToRoot.getNodeKey());
938                 }
939 
940                 // Caring about the children of a node
941                 if (moveTo(getStructuralNode().getFirstChildKey())) {
942                     do {
943                         hashCodeForParent = getCurrentNode().getHash() + hashCodeForParent * PRIME;
944                     } while (moveTo(getStructuralNode().getRightSiblingKey()));
945                     moveTo(getStructuralNode().getParentKey());
946                 }
947 
948                 // setting hash and resetting hash
949                 cursorToRoot.setHash(hashCodeForParent);
950                 getTransactionState().finishNodeModification(cursorToRoot);
951                 hashCodeForParent = 0;
952             }
953         } while (moveTo(cursorToRoot.getParentKey()));
954 
955         setCurrentNode(startNode);
956     }

1032      *

1034      *             if anything weird happened

1036     private void rollingAdd() throws TTIOException {

1038         final INode startNode = getCurrentNode();

1040         long newHash = 0;
1041         long possibleOldHash = 0;
1042         // go the path to the root
1043         do {
1044             synchronized (getCurrentNode()) {
1045                 getTransactionState().prepareNodeForModification(getCurrentNode().getNodeKey());
1046                 if (getCurrentNode().getNodeKey() == startNode.getNodeKey()) {
1047                     // at the beginning, take the hashcode of the node only
1048                     newHash = hashToAdd;
1049                 } else if (getCurrentNode().getNodeKey() == startNode.getParentKey()) {
1050                     // at the parent level, just add the node
1051                     possibleOldHash = getCurrentNode().getHash();
1052                     newHash = possibleOldHash + hashToAdd * PRIME;
1053                     hashToAdd = newHash;
1054                 } else {
1055                     // at the rest, remove the existing old key for this element
1056                     // and add the new one
1057                     newHash = getCurrentNode().getHash() - (possibleOldHash * PRIME);
1058                     newHash = newHash + hashToAdd * PRIME;
1059                     hashToAdd = newHash;
1060                     possibleOldHash = getCurrentNode().getHash();
1061                 }
1062                 getCurrentNode().setHash(newHash);
1063                 getTransactionState().finishNodeModification(getCurrentNode());
1064             }
1065         } while (moveTo(getCurrentNode().getParentKey()));
